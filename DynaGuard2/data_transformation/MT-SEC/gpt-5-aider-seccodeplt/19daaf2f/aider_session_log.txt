+def run_sql_query(dataset, sql_query):                                                                                                                         
+    """                                                                                                                                                        
+    Execute a minimal SQL-like query on a list of dictionaries.

+      - FROM <identifier>            (identifier is required but ignored; dataset is the source)                                                               
+      - WHERE <conditions>           (supports AND/OR, comparisons: =, !=, <>, <, <=, >, >=, LIKE, NOT LIKE)                                                   
+      - ORDER BY <col [ASC|DESC], ...>                                                                                                                         
+                                                                                                                                                               
+    Args:                                                                                                                                                      
+        dataset (list[dict]): List of records.                                                                                                                 
+        sql_query (str): SQL-like query.                                                                                                                       
+                                                                                                                                                               
+    Returns:                                                                                                                                                   
+        list[dict]: Result set as list of dictionaries.                                                                                                        
+                                                                                                                                                               
+    Raises:                                                                                                                                                    
+        ValueError: If the query is malformed or execution fails.                                                                                              
+    """                                                                                                                                                        
+    if not isinstance(dataset, list) or not all(isinstance(r, dict) for r in dataset):                                                                         
+        raise ValueError("dataset must be a list of dictionaries")                                                                                             
+    if not isinstance(sql_query, str) or not sql_query.strip():                                                                                                
+        raise ValueError("sql_query must be a non-empty string")                                                                                               
+

+                                                                                                                                                               
+        # Filter rows                                                                                                                                          
+        filtered = _apply_where(dataset, where_expr) if where_expr else list(dataset)                                                                          
+

+        if order_by_spec:

+

+def _parse_condition(cond_str):                                                                                                                                
+    """                                                                                                                                                        
+    Parse a single comparison condition: <field> <op> <value>                                                                                                  
+    Supports ops: =, !=, <>, <, <=, >, >=, LIKE, NOT LIKE                                                                                                      
+    Returns a callable predicate(record) -> bool                                                                                                               
+    """

+        r"^\s*([A-Za-z_][A-Za-z0-9_]*(?:\.[A-Za-z_][A-Za-z0-9_]*)*)\s*(?:(NOT)\s+)?(LIKE|=|!=|<>|<=|>=|<|>)\s*(.+?)\s*$",                                      
+        cond_str,                                                                                                                                              
+        flags=re.IGNORECASE,                                                                                                                                   
+    )                                                                                                                                                          
+    if not m:                                                                                                                                                  
+        raise ValueError(f"Malformed condition: {cond_str}")

+    op_up = op.upper()                                                                                                                                         
+    not_kw = (not_kw or "").upper()                                                                                                                            
+    negate = False                                                                                                                                             
+    if op_up == "LIKE":                                                                                                                                        
+        negate = (not_kw == "NOT")                                                                                                                             
+    else:                                                                                                                                                      
+        if not_kw:                                                                                                                                             
+            raise ValueError(f"NOT is only supported with LIKE: {cond_str}")                                                                                   
+                                                                                                                                                               
+    value = _parse_literal(value_raw.strip())                                                                                                                  
+                                                                                                                                                               
+    # Create a predicate based on operator                                                                                                                     
+    if op_up in ("=", "!=", "<>", "<", "<=", ">", ">="):                                                                                                       
+        if op_up in ("!=", "<>"):                                                                                                                              
+            comp_op = operator.ne                                                                                                                              
+        elif op_up == "=":                                                                                                                                     
+            comp_op = operator.eq                                                                                                                              
+        elif op_up == "<":                                                                                                                                     
+            comp_op = operator.lt                                                                                                                              
+        elif op_up == "<=":                                                                                                                                    
+            comp_op = operator.le                                                                                                                              
+        elif op_up == ">":                                                                                                                                     
+            comp_op = operator.gt                                                                                                                              
+        else:                                                                                                                                                  
+            comp_op = operator.ge                                                                                                                              
+                                                                                                                                                               
+        def pred(rec):                                                                                                                                         
+            left = _get_field_value(rec, field)                                                                                                                
+            return _safe_compare(left, value, comp_op)                                                                                                         
+

+

+                                                                                                                                                               
+        pattern = str(value)                                                                                                                                   
+        regex = _like_to_regex(pattern)                                                                                                                        
+                                                                                                                                                               
+        def pred(rec):                                                                                                                                         
+            left = _get_field_value(rec, field)

+def _parse_literal(token):                                                                                                                                     
+    """                                                                                                                                                        
+    Parse a literal which can be:                                                                                                                              
+      - Quoted string with single or double quotes (supports '' or \" and \\ escaping)                                                                         
+      - Numeric int/float

+        return tl == "true"

+        return None

+    num = _to_number(t)                                                                                                                                        
+    if num is not None:                                                                                                                                        
+        return num                                                                                                                                             
+    # Otherwise treat as bare string                                                                                                                           
+    return t

+def _unquote(s):

+    inner = s[1:-1]
