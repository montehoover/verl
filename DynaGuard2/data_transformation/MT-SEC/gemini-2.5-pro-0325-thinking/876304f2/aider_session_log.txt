    # ^ asserts position at start of the string.

    # @ matches the literal "@" character.                                                                            
    # ([a-zA-Z0-9.-]+\.[a-zA-Z]{2,}) captures the domain part:                                                        
    #   [a-zA-Z0-9.-]+ matches domain name parts (e.g., "example", "sub.domain").
