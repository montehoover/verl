ALLOWED_NODES = {

    ast.Num,       # For numbers in Python < 3.8                                                                      
    ast.BinOp,                                                                                                        
    ast.UnaryOp,                                                                                                      
    ast.Add,                                                                                                          
    ast.Sub,                                                                                                          
    ast.Mult,                                                                                                         
    ast.Div,

    ast.Mod,       # Modulo                                                                                           
    ast.FloorDiv,  # Integer division                                                                                 
    ast.USub,      # Unary minus                                                                                      
    ast.UAdd,      # Unary plus (less common but valid)

def _is_node_safe(node):                                                                                              
    """                                                                                                               
    Recursively checks if an AST node and its children are in the ALLOWED_NODES whitelist.                            
    """                                                                                                               
    if type(node) not in ALLOWED_NODES:                                                                               
        # For ast.Constant, also check the type of the value if it's a number                                         
        if isinstance(node, ast.Constant) and not isinstance(node.value, (int, float, complex)):                      
            return False                                                                                              
        elif not isinstance(node, ast.Constant): # if not a Constant, and not in ALLOWED_NODES, it's unsafe           
             return False                                                                                             
                                                                                                                      
                                                                                                                      
    for child_node in ast.iter_child_nodes(node):                                                                     
        if not _is_node_safe(child_node):                                                                             
            return False                                                                                              
    return True
