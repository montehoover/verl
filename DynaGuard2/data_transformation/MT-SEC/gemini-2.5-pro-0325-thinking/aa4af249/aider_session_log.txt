def _parse_condition(condition_str):

    into (field, operator_function, typed_value).

    # Regex to capture field, operator, and value. Value can be quoted or unquoted.

    if not match:                                                                                                     
        raise ValueError(f"Invalid WHERE condition format: {condition_str}")                                          
                                                                                                                      
    field, op_str, value_literal = match.groups()                                                                     
    value_literal = value_literal.strip()                                                                             
                                                                                                                      
    op_func = _OPERATORS.get(op_str)                                                                                  
    if not op_func:                                                                                                   
        raise ValueError(f"Unsupported operator: {op_str}")                                                           
                                                                                                                      
    # Type determination for value_literal                                                                            
    typed_value = None                                                                                                
    if (value_literal.startswith("'") and value_literal.endswith("'")) or \                                           
       (value_literal.startswith('"') and value_literal.endswith('"')):                                               
        typed_value = value_literal[1:-1]  # String type                                                              
    else:

                    f"but record has {type(record_value).__name__} type."

        raise ValueError("Invalid SQL query: No columns specified in SELECT clause.")
